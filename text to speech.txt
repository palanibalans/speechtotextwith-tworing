		function textToAudio() {
                let msg = document.getElementById("text-to-speech").value;
                
                let speech = new SpeechSynthesisUtterance();
				var voices = window.speechSynthesis.getVoices();
                speech.lang = "en-US";
				
				
                
                speech.text = msg;
                speech.volume = 1;
                speech.rate = 1;
                speech.pitch = 2;
               
                window.speechSynthesis.speak(speech);
            }
